CREATE VIEW `markerviews` AS select `m`.`id` AS `id`,`m`.`category_id` AS `category_id`,`m`.`user_id` AS `user_id`,`m`.`respondent_id` AS `respondent_id`,`m`.`weather_id` AS `weather_id`,`m`.`lat` AS `lat`,`m`.`lng` AS `lng`,`m`.`created` AS `created`,`m`.`modified` AS `modified`,`m`.`info` AS `info`,`r`.`twitUserID` AS `twitUserID`,`m`.`twitID` AS `twitID`,`m`.`twitTime` AS `twitTime`,`m`.`twitURL` AS `twitURL`,`m`.`twitPlaceID` AS `twitPlaceID`,`m`.`twitPlaceName` AS `twitPlaceName`,`m`.`isTwitPlacePrecise` AS `isTwitPlacePrecise`,`m`.`twitImage` AS `twitImage`,`m`.`pinned` AS `pinned`,`m`.`cleared` AS `cleared`,`m`.`active` AS `active`,`c`.`name` AS `category_name`,`u`.`username` AS `username`,`u`.`email` AS `user_email`,`r`.`name` AS `respondent_name`,`r`.`contact` AS `respondent_contact`,`w`.`name` AS `weather_name`,`p`.`name` AS `place_name`,if((`r`.`twitUserID` is not null),1,0) AS `isTwitUserIDExist`,if((`m`.`twitID` is not null),1,0) AS `isTwitExist`,if((`m`.`twitImage` is not null),1,0) AS `isTwitImageExist`,if((`m`.`twitURL` is not null),1,0) AS `isTwitURLExist`,if((`p`.`name` is not null),1,0) AS `isPlaceNameExist` from (((((`markers` `m` left join `categories` `c` on((`m`.`category_id` = `c`.`id`))) left join `users` `u` on((`m`.`user_id` = `u`.`id`))) left join `respondents` `r` on((`m`.`respondent_id` = `r`.`id`))) left join `weather` `w` on((`m`.`weather_id` = `w`.`id`))) left join `places` `p` on(((`m`.`lat` = `p`.`lat`) and (`m`.`lng` = `p`.`lng`))))
